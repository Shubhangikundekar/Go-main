Slip 20
B) Write a program in Go language how to create a channel and
illustrate how to close a channel using for range loop and close
function.
package main
import "fmt"

func main() {

	ch := make(chan int)

	go func() {
		for i := 1; i <= 5; i++ {
			ch <- i
		}
		close(ch)
	}()

	for val := range ch {
		fmt.Println(val)
	}
}
