Q1. A) Write a program in GO language to print a recursive sum of digits [20 Marks]
of a given number.
package main

import "fmt"


func sumOfDigits(num int) int {
	if num == 0 {
		return 0
	}
	return (num % 10) + sumOfDigits(num/10)
}

func main() {
	var number int

	fmt.Print("Enter a number: ")
	fmt.Scanln(&number)

	
	if number < 0 {
		number = -number
	}

	result := sumOfDigits(number)

	fmt.Println("Sum of digits:", result)
}
