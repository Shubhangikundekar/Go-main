Q1. A) Write a program in GO language to print a multiplication table of [20 Marks]
number using function.
package main

import "fmt"


func printTable(num int) {
    for i := 1; i <= 10; i++ {
        fmt.Printf("%d x %d = %d\n", num, i, num*i)
    }
}

func main() {
    var number int

    fmt.Print("Enter a number: ")
    fmt.Scan(&number)

       printTable(number)
}
