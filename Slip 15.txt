package main

import "fmt"

func calculate(a int, b int) (int, int, int) {
    sum := a + b
    diff := a - b
    product := a * b
    return sum, diff, product
}

func main() {
    var x, y int

    fmt.Print("Enter two numbers: ")
    fmt.Scan(&x, &y)

   
    s, d, p := calculate(x, y)

    fmt.Println("Sum:", s)
    fmt.Println("Difference:", d)
    fmt.Println("Product:", p)
}
